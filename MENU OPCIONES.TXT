CREATE OR REPLACE FORCE VIEW VW_PROCESOS_USUARIO AS 
SELECT P.ID AS IDPROCESO, U.ID AS IDUSUARIO,U.USUARIO, r.descripcion AS AUTORIZACION, p.descripcion AS PROCESO 
FROM usuarios U, perfiles_digitalizacion P, roles r 
WHERE r.perfil_usuario= u.id and p.id= r.perfil_digitalizacion
